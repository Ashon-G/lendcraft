<HowToStep index='3' title="Connect the data layer">

The preview ships with mocked quest data. Replace the `useDemoQuests` hook with your own GraphQL or REST fetcher and the UI will update instantly thanks to Expo Fast Refresh.

<CodeBlock fileName="hooks/useDemoQuests.ts">

```ts
import { useEffect, useState } from 'react';

export function useDemoQuests() {
  const [quests, setQuests] = useState(initialQuests);

  useEffect(() => {
    const interval = setInterval(() => {
      setQuests((current) =>
        current.map((quest) =>
          quest.id === 'weekly'
            ? { ...quest, progress: Math.min(quest.progress + 10, 100) }
            : quest
        )
      );
    }, 4000);

    return () => clearInterval(interval);
  }, []);

  return { quests };
}
```

</CodeBlock>
